Ques 1. Find specific objects in an array of objects using JavaScript .

Ans: For that, I created an array of objects representing employees and used the filter method to find employees based on a specific condition.

Example:

const employees = [
  { name: "Sourav", from: "Noida" },
  { name: "Mudgal", from: "Delhi" },
  { name: "Ketan", from: "Lucknow" },
  { name: "Shukla", from: "Lucknow" },
  { name: "Bippan",from: "Noida" },
];
function findEmployeeFrom() { return employees.filter((employee) => employee.from === 'Noida'); }
const result = findEmployeeFrom(); console.log(result);

Output: [ { "name": "Sourav", "from": "Noida" }, { "name": "Bippan", "from": "Noida" } ]
