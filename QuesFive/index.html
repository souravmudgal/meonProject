How to control the duration of an Ajax request? Give Example

Ans: To control the duration of the Ajax request i used the xhr.timeout The timeout property in XMLHttpRequest is intended to set the maximum time allowed for the entire request to complete, including the time it takes for the server to respond.

Example:

"use strict";
var xhr = new XMLHttpRequest();
xhr.open('GET', `https://restcountries.com/v3.1/name/india?fullText=true`, true);
xhr.timeout = 100000;
xhr.onload = function ()
{
if (xhr.status >= 200 && xhr.status < 300) {
console.log('Ajax request successful:', xhr.responseText); }
else
{ console.error('Ajax request failed:', xhr.statusText); } };
// Handle timeout event
xhr.ontimeout = function ()
{ console.error('Ajax request timed out'); };
// Handle other errors
xhr.onerror = function () { console.error('Ajax request failed'); };
xhr.send();
